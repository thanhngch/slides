Sequelize: query where which Op

Project.findAll({
    where: {
        id: {
        [Op.and]: {a: 5},           // AND (a = 5)
        [Op.or]: [{a: 5}, {a: 6}],  // (a = 5 OR a = 6)
        [Op.gt]: 6,                // id > 6
        [Op.gte]: 6,               // id >= 6
        [Op.lt]: 10,               // id < 10
        [Op.lte]: 10,              // id <= 10
    }
})

Sequelize: order by

db.Page.findAll({
    include: [{
        model: db.Gallery
        include: [{
            model: db.Artwork
        }]
    }],
    order: [
        [ db.Gallery, 'order', 'DESC' ],
        [ db.Gallery, db.ArtWork, 'order', 'DESC' ]
    ]
})
