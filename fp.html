// JS
function fibonacci (n) {
    if (n < 0) {
        throw new Error("Invalid argument");
    }
    if (n == 0 || n == 1) {
        return 1;
    }
    return fibonacci (n - 1) + fibonacci (n - 2);
}

Currying

Curry is important thing in Functional Programing

Currying (next)


// JS
add = function (x, y) {
    return x + y
}
                    

// arrow way
add = (x, y) => x + y

Currying (next)


// JS
add2 = function (x) {
    return function (y) {
        return x + y
    }
}
                    

// arrow way
add2 = x => y => x + y

reduce


// JS
const num = [1, 2, 3, 4];
const total_reducer = (acc, value) => acc + value;
num.reduce(total_reducer) // return 10
                    

-- Haskell

Filter


// JS
var words = ['spray', 'limit', 'elite', 
    'exuberant', 'destruction', 'present'];

const result = words.filter(word => word.length > 6);
// result = ["exuberant", "destruction", "present"]

let words = ["spray", "limit", "elite", "exuberant",
    "destruction", "present"]
filter ((> 6) . length) words
-- result = ["exuberant", "destruction", "present"]

Composition

(f.g)(x) = f(g(x))

Composition (next)


// JS
const users = [
    { name: "Jeff",   age: 14 },
    { name: "Jack",   age: 18 }, 
    { name: "Milady", age: 22 },
]
const filter = (cb, arr) => arr.filter(cb);
const map = (cb, arr) => arr.map(cb);

map(u => u.name, filter(u => u.age > 16, users));
// return ["Jack", "Milady"]

Composition (next)


compose(function1, function2, ... , functionN): Function
                    

// JS
const compose = (...functions) => args => 
    functions.reduceRight((arg, fn) => fn(arg), args);

const filter = cb => arr => arr.filter(cb);
const map = cb => arr => arr.map(cb);

compose(
    map(u => u.name),
    filter(u => u.age > 16)
)(users) 
// return ["Jack", "Milady"]

Composition (next)


-- Haskell
let users = [ ("Jeff", 14 ), 
    ("Jack", 18 ), ("Milady", 22 )]

-- Solution
map (\(name, _) -> name) 
    (filter (\(_, age) -> age > 16) users)

-- Another way using composition
(map (\(name, _) -> name) . 
    filter (\(_, age) -> age > 16)) users
-- return ["Jack", "Milady"]
